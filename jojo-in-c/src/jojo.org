    void
    destroy_env (env_t **self_pt)
    {
        assert (self_pt);
        if (*self_pt) {
            env_t *self = *self_pt;
            destroy_name_dict (&self->name_dict);
            destroy_data_stack (&self->data_stack);
            destroy_frame_stack (&self->frame_stack);
            free (self);
            *self_pt = NULL;
        }
    }
